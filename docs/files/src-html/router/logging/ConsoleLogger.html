016    package router.logging;
017    
018    import java.io.FileWriter;
019    import java.util.Calendar;
020    
021    /**
022     * The <code>ConsoleLogger</code> class provides a logger that outputs to the console.
023     */
024    public class ConsoleLogger extends Logger {
025            
026        private FileWriter writer;
027        private boolean isClosed = false;
028    
029        
030        
031        /**
032         * Create a <code>ConsoleLogger</code> for initialisation 
033         */
034        public ConsoleLogger() {
035            super(0);
036            write("Console Logger created for initialisation.");
037        }
038        
039        
040        
041        /**
042         * Create a <code>ConsoleLogger</code> with the specified router ID
043         * @param routerId The router ID to use
044         */
045        public ConsoleLogger(int routerId) {
046            super(routerId);
047            write("Console Logger created for router-"+routerId);
048        }
049        
050        
051        
052        /**
053         * Write to the logger
054         * @param str The string to write
055         */
056        @Override
057        public synchronized void write(String str) {
058            String time = String.format("%tT : ", Calendar.getInstance());
059            System.out.print(time+str+"\n");
060        }
061    
062        
063        
064        /**
065         * Close the logger and the associated file
066         */
067        @Override
068        public synchronized void close() {}
069    
070        
071        
072        
073    
074    }
